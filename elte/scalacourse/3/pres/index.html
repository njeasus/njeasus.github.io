Pattern matching user tipusokra

A matchben nincs semmi kulonleges

def apply(part1: A, part2: B): Construct

A case classok automatikusan adjak

class Person private (val firstName: String, val surName: String)
object Person {
    def apply(firstName: String, surName: String) = new Person(firstName, surName)
    def unapply(p: Person): Option[(String, String)] = Some((p.firstName, p.surName))
}

sealed trait IntTree {
    def depthFirst(op: IntTree => Unit): Unit
}
case object EmptyIntTree extends IntTree {
    override def depthFirst(op: IntTree => Unit): Unit = {}
}
case class IntNode(v: Int, left: IntTree = EmptyIntTree, right: IntTree = EmptyIntTree) extends IntTree {
    override def depthFirst(op: IntTree => Unit): Unit = {
      left.depthFirst(op)
      right.depthFirst(op)
      op(this)
    }
}
