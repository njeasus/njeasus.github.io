import shapeless._

// tipushelyes heterogen lista
val sets = Set(1d, 2d, 3d) :: Set("4", "5", "6") :: HNil

// polimorfikus fuggveny: Setekrol Optionokre kepez
object headOption extends (Set ~> Option) {
    def apply[T](set: Set[T]): Option[T] = set.headOption
}

// map over hlist

Recordok

import shapeless._
import syntax.singleton._
import record._
import shapeless.test._

// Singleton literal type - az adott literalbol kepez egy uj tipust: itt Int(23) es Int(24)
// Ezek a tipusok nevei, nem konstruktor hivasok
// sameTyped: makro, ami a ket kifejezes tipusat hasonlitja ossze, es forditasi hibat dob, ha nem egyeznek 
// sameTyped(23.narrow)(24.narrow)

// Record: key -> value parok heterogen listaja (ez ugye igazabol egy heterogen map)
// Kulcsok singleton typed literalok => tipushelyes reprezentacio
val person = ("name" ->> "John Doe") :: ("age" ->> 23) :: HNil

// Compile-time hiba, nincs ilyen kulcs
// println(person("address"))
// Mint egy map:
person.keys
person.values

// Ertek update egy mar letezo kulcsra
val aYearLater = person + ("age" ->> 24)
// Lekerdezheto a kulcshoz tartozo ertek
println(aYearLater("age"))

// Kulcs (es az erteke) eltavolitasa
val personWithNameOnly = person - "age"
// Compile-time hiba
// personWithNameOnly("age")
println(personWithNameOnly("name"))

Lencsek

import shapeless._

case class Address(street : String, city : String, postcode : String)
case class Person(name : String, age : Int, address : Address)

val person = Person("John Doe", 23, Address("Main", "Hill Valley", "H1234"))
println(person)
// Case class-ok modositasa copyn keresztul: elegge korulmenyes, es itt csak 2 szint van meg
val movedPerson = person.copy(address = person.address.copy(street = "OffMain"))
println(movedPerson)

// Lencse, ami rafokuszal egy struktura egy adott elemere
// A lencse ujrahasznosithato. Kb mint egy query az adott strukturan belul (pl XPath XMLnel)
// Akar el is menthetjuk a hasznos lencseket egy companion objectbe
val streetLens = lens[Person] >> 'address >> 'street

// Lekerdezes lencsevel
println(streetLens.get(person))
println(streetLens.get(movedPerson))

// Modositas lencsevel
val movedPerson2 = streetLens.set(person)("OffMain")
println(movedPerson2)
assert(movedPerson == movedPerson2)

Equal, Enum es type tageles

import scalaz._
import Scalaz._
import syntax.equal._

// Hasonlo makro, mint a sameTyped
//  1 === 2.0

// Enumeralt ertekek generalasa
val numbers = 3 |=> 5 // stream
val numbers2 = 3 |-> 5 // list
// stream - lazy, list - concrete
println(numbers.toList)
println(numbers2.toList)

// Type tagging
// Tipusok hasznalat szerinti megkulonboztetese. Pl. mertekegysegek: alapvetoen egy szam ertek, de
// ket Double ket nagyon kulonobozo dolgot jelent, pl. tomeg es sebesseg
// Type alias mukodhet(ne), de nagyon korulmenyes, pl type KiloGramDouble = Double; type KiloGramInt = Int
// Ehelyett "ad-hoc" tageljuk a tipust
sealed trait KiloGram
// Ez maga a tagelo fuggveny, egy A-bol csinal egy A+Kilogram tipust
def KiloGram[A](a: A): A @@ KiloGram = Tag[A, KiloGram](a)
// Ezzel maris tudunk Double-lel tagelni
val mass: Double @@ KiloGram = KiloGram(20.0)

// A felhasznalas helyen megmondjuk, hogy szuksegunk van egy Int ertekre, ami KiloGramot reprezental
def decreaseKiloGram(value: Int @@ KiloGram): Int @@ KiloGram = 
    KiloGram(Tag.unwrap(value) - 1) // A Tag.unwrap adja vissza a valodi erteket

println(decreaseKiloGram(KiloGram(10)))

Memo

Referentially transparent: egy adott bemeneti ertekre mindig ugyanazt a kimenetet adja vissza

RT: szimbolikusan helyettesitheto az ertekkel

Memoizacio: fuggveny bemenetehez a kalkulalt ertek megjegyzese es hasznalata

def doubleComplex(input: Int): Int = {
    // Szimulaljunk egy bonyolult szamitast
    Thread.sleep(5000)
    input * input
}

// Utility megmerni, hogy mennyi ideig futott egy fuggveny
def timed[U](work: => U): U = {
    val startTime = System.currentTimeMillis()
    val data = work
    val endTime = System.currentTimeMillis()
    println(s"Timed(${endTime - startTime}): ${data}")
    data
}

// Memo keszitese a fuggvenyunkbol
val longRunningMemo = Memo.immutableHashMapMemo(doubleComplex)
timed(longRunningMemo(1)) // 5000ms
timed(longRunningMemo(1)) // 0ms
timed(longRunningMemo(2)) // 5000ms
timed(longRunningMemo(2)) // 0ms

Free monadok

A monadok egymastol fuggo szamitasokat reprezentalnak, amik a monadon belul maradnak

Free monad: monad, ami atalkithato barmilyen mas monadda

Funktor: egy fuggveny, ami atalakit egy erteket egy belso kontexten belul: unwrap-apply-wrap

SQL tranzakcio free monaddal

case class SqlOp[A](op: Connection => A) // Ez egy SQL muvelet
// A SQL muvelet Functor, ami vegrehajtja a muveletet
implicit object SqlOpFunctor extends Functor[SqlOp] {
    def map[A,B](a: SqlOp[A])(g: A => B) = SqlOp[B]((c:Connection) => g(a.op(c)))
}
type Sql[A] = Free[SqlOp,A] // A free monadunk

// A kulonbozo SqlOp-ok kiemelese free monadda - ezek nem commitolhatnak!
def sqlop1(): Sql[Boolean] = Free.liftF(SqlOp((conn:java.sql.Connection) => {
    println("Running SQL OP1")
    true
}))
def sqlop2(): Sql[Boolean] = Free.liftF(SqlOp((conn:java.sql.Connection) => {
    println("Running SQL OP2")
    true
}))
def emptysqlop(): Sql[Boolean] = Free.liftF(SqlOp((conn:java.sql.Connection) => {
    println("Running Empty SQL")
    true
}))

// A tranzakcionk SqlOp-ok kompozicioja
def myTransaction(): Sql[Boolean] = for {
    condition <- sqlop1()
    _ <- if (condition) sqlop2() else emptysqlop()
} yield condition

myTransaction() // nem futtat meg le semmit, kell egy interpreter, ami tudja, hogyan kell banni a monadunkkal
  
// Az interpreter - az implicit funktor itt kerul atadasra
def runTransactionImpl[A](conn: Connection, trans: Sql[A]): A = trans.fold(
    (a: A) => a, // egy muvelettel mit kell kezdeni (=pointnak is hivjak)
    (x: SqlOp[Sql[A]]) => runTransactionImpl(conn, x.op(conn)) // hogyan kell tovabbhaladni a kovetkezo muveletre
)

// Itt valoban vegrehajtjuk a tranzakciot
runTransactionImpl(DB.connect(), myTransaction())
